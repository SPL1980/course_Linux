第八章 shell编程

变量

num=2
echo "this is the ${num}nd" # this is the 2nd
VAR=1
VAR=$VAR+1
echo $VAR
echo $SHELL # 输出:/bin/bash
echo "$SHELL" # 同上
echo '$SHELL' # 输出:$SHELL
echo \$SHELL # 同上

算术运算

var=1 # var:1
let "var+=1" # var:2
var="$[$var+1]" # var:3
((var++)) # var:4
var=$(($var+1)) # var:5
var="$(expr "$var" + 1)" # var:6,不建议使用
var="`expr "$var" + 1`" # var:7,强烈不建议使用,注意加号两边的空格,否则还是按照字符串的方式赋值,`为Esc下方的`,而不是单引号'
